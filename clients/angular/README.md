# MyMusik Angular API Client

This folder contains the generated Angular/TypeScript API client for the MyMusik music-player API.

Source included here:

```text
src/api
```

The code was generated by `ng-openapi-gen` and is designed to be copied into an Angular application. The default API root URL is:

```text
http://localhost:7790/api
```

## Use In Angular

Import the generated API module or generated services from `src/api`, then configure the API root URL:

```ts
import { provideApiConfiguration } from './api/api-configuration';
import { HttpInterceptorFn, provideHttpClient, withInterceptors } from '@angular/common/http';

const mymusikAuthInterceptor: HttpInterceptorFn = (req, next) => {
  const token = localStorage.getItem('mymusik_access_token');
  return next(token
    ? req.clone({ setHeaders: { Authorization: `Bearer ${token}` } })
    : req
  );
};

export const appConfig = {
  providers: [
    provideApiConfiguration('http://localhost:7790/api'),
    provideHttpClient(withInterceptors([mymusikAuthInterceptor]))
  ]
};
```

If you use a different auth store, keep the same HTTP header shape:

```http
Authorization: Bearer YOUR_ACCESS_TOKEN
```

## Player Example

This service fetches real `Song` objects from the API, then reuses those generated objects for playback and queue operations.

```ts
import { Injectable } from '@angular/core';
import { switchMap } from 'rxjs';
import { PlayerService, QueueItemService, SongService } from './api/services';
import { SongList } from './api/models';

@Injectable({ providedIn: 'root' })
export class MyMusikPlayerExample {
  constructor(
    private readonly songs: SongService,
    private readonly player: PlayerService,
    private readonly queue: QueueItemService
  ) {}

  playFirstThreeSongs() {
    return this.songs.getPage({ sortBy: 'name', orderBy: 'ASC', pageIndex: 0 }).pipe(
      switchMap((songs) => {
        const body: SongList = { id: '', songs: songs.slice(0, 3) };
        return this.player.playSongs({ body });
      })
    );
  }

  addNewestSongsToQueue() {
    return this.songs.getPage({ sortBy: 'created', orderBy: 'DESC', pageIndex: 0 }).pipe(
      switchMap((songs) => {
        const body: SongList = { id: '', songs: songs.slice(0, 10) };
        return this.queue.addSongs({ body });
      })
    );
  }

  togglePlayback() {
    return this.player.startStopTogglePlaying();
  }

  nextTrack() {
    return this.player.nextTrack();
  }

  clearQueue() {
    return this.queue.clear();
  }
}
```

A tiny component can call it like this:

```ts
import { Component } from '@angular/core';
import { MyMusikPlayerExample } from './mymusik-player-example.service';

@Component({
  selector: 'app-player-controls',
  template: `
    <button type="button" (click)="play()">Play first songs</button>
    <button type="button" (click)="queue()">Queue newest</button>
    <button type="button" (click)="toggle()">Play/Pause</button>
    <button type="button" (click)="next()">Next</button>
  `
})
export class PlayerControlsComponent {
  constructor(private readonly mymusik: MyMusikPlayerExample) {}

  play() {
    this.mymusik.playFirstThreeSongs().subscribe();
  }

  queue() {
    this.mymusik.addNewestSongsToQueue().subscribe();
  }

  toggle() {
    this.mymusik.togglePlayback().subscribe();
  }

  next() {
    this.mymusik.nextTrack().subscribe();
  }
}
```
